1. SETUP STATIC IP
	sudo nano /etc/network/interfaces

		#################################
		# BEGIN Network Interface File  #
		#################################

		#################################
		# Loopback Interface            #
		#################################

		auto lo
		iface lo inet loopback

		#################################
		# WAN - p5p1                    #
		#################################

		auto p5p1
		iface p5p1 inet static
		address 208.185.141.45
		netmask 255.255.255.248
		gateway 208.185.141.41
		dns-nameservers 8.8.8.8 8.8.4.4

		#################################
		# LAN - p4p1                    #
		#################################

		auto p4p1
		iface p4p1 inet static
		address 192.168.1.2
		netmask 255.255.255.0

		#################################
		# END Network Interfaces        #
		#################################

	sudo /etc/init.d/networking restart

2. MOVE TO REMOTE MACHINE

3. SETUP SSH SECURITY
	sudo apt-get install openssh-server
	sudo cp /etc/ssh/sshd_config /etc/ssh/sshd_config.factory-defaults
	sudo chmod a-w /etc/ssh/sshd_config.factory-defaults
	
		#### SSH Server System Configuration File ####

		#### Listening Ports/IPs
		Port 8822

		#### Use these options to restrict which interfaces/protocols sshd will bind to
		Protocol 2

		#### HostKeys for protocol version 2
		HostKey /etc/ssh/ssh_host_rsa_key
		HostKey /etc/ssh/ssh_host_dsa_key
		HostKey /etc/ssh/ssh_host_ecdsa_key
		HostKey /etc/ssh/ssh_host_ed25519_key

		#### Privileges
		UsePrivilegeSeparation yes

		#### Lifetime and Size of Generated Server Key
		KeyRegenerationInterval 3600
		ServerKeyBits 1024

		#### Logging
		SyslogFacility AUTH
		LogLevel INFO

		#### Authentication:
		LoginGraceTime 45
		PermitRootLogin without-password
		StrictModes yes

		#### SSH Key Authentication:
		RSAAuthentication yes
		PubkeyAuthentication yes
		AuthorizedKeysFile /etc/ssh/%u/authorized_keys

		#### Rhosts Authentication:
		IgnoreRhosts yes
		RhostsRSAAuthentication no

		#### Protocol V2 Authentication:
		HostbasedAuthentication no

		#### Authentication Security:
		PermitEmptyPasswords no
		ChallengeResponseAuthentication no
		PasswordAuthentication yes

		#### Forwarding
		AllowTcpForwarding no
		X11Forwarding no

		#### Banner
		PrintMotd yes
		PrintLastLog yes
		Banner /etc/ssh/arcadmin/banner

		TCPKeepAlive yes
		#UseLogin no
		#MaxStartups 10:30:60

		#### Client Locale ENV vars
		AcceptEnv LANG LC_*
		Subsystem sftp /usr/lib/openssh/sftp-server

		#### END SSH Server System Configuration File ####

4. CREATE SSH KEYS ON CLIENT MACHINE

5. SETUP KEYS DIRECTORY
	
	sudo mkdir /etc/ssh/<username>
	sudo chmod 755 /etc/ssh/<username>

6. COPY PUBLIC KEY TO SERVER

7. SETUP SSH "authorized_keys" FILE

	sudo su
	ssh-keygen -i -f <COPIED KEY FILENAME.pub> >> /etc/ssh/<username>/authorized_keys
	chmod 644 /etc/ssh/<username>/authorized_keys
